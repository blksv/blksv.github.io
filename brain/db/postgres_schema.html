with foreign_keys as (
    select
        conrelid::regclass::text AS table_name, pa.attname as column_name,
        confrelid::regclass::text AS foreign_table_name, paf.attname as foreign_column_name
    from pg_constraint pc
        left join pg_attribute pa
            on pc.conrelid = pa.attrelid
                and conkey[1] = pa.attnum
        left join pg_attribute paf
            on pc.confrelid = paf.attrelid
                and confkey[1] = paf.attnum
    where contype='f'
)
select jsonb_object_agg(r.table_name, r.columns)
from (
    select table_name, array_agg(to_json(c)) as columns
    from (
        select c.table_name, c.column_name, c.udt_name,
            c.is_nullable = 'YES' as is_nullable,
            c.is_generated = 'ALWAYS' as is_generated,
            c.is_updatable = 'YES' as is_updatable,
            c.column_default is not null as has_default,
            fk.foreign_table_name, fk.foreign_column_name,
            pgd.description
        from information_schema.columns c
            left join foreign_keys fk
                on fk.table_name = c.table_name
                    and fk.column_name = c.column_name
            left join pg_catalog.pg_description pgd
                on pgd.objoid=c.table_name::regclass::oid
                    and pgd.objsubid=c.ordinal_position
        where table_schema = 'public'
        order by c.table_name, c.ordinal_position
    ) c
    group by table_name
) r
