import jinja2.ext

class CustomTag(jinja2.ext.Extension):
    def __init__(self, env):
        super().__init__(env)

    def _run(self, *args, caller=None):
        if self.with_content:
            return self.__class__.run(args[0], caller(), *args[1:])
        return self.__class__.run(args[0], *args[1:])

    def parse(self, parser):
        lineno = next(parser.stream).lineno
        context = jinja2.nodes.ContextReference()
        args = [context]
        while not parser.stream.current.test('block_end'):
            args.append(parser.parse_expression())
        if self.with_content:
            body = parser.parse_statements([f'name:end{self.tags[0]}'], drop_needle=True)
            return jinja2.nodes.CallBlock(self.call_method('_run', args), [], [], body).set_lineno(lineno)
        call = self.call_method('_run', args, lineno=lineno)
        return jinja2.nodes.Output(
            [call],
            lineno=lineno
        )


def add_tag(env, tag, func, with_content=False):
    ext = type(f"{tag}Extension", (CustomTag,), {})
    ext.tags = [tag]
    ext.run = func
    ext.with_content = with_content
    env.add_extension(ext)
