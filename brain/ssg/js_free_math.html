KaTeX node.js math rendering server

const http = require('http');
const katex = require('katex');

const katexOpt = {
    output: 'html',
    throwOnError: 'false'
}

function readBody(req) {
    ret = new Promise((resolve, reject) => {
        const body = [];
        req.on('error', (err) => {
            reject(err);
        }).on('data', (chunk) => {
            body.push(chunk);
        }).on('end', () => {
            resolve(Buffer.concat(body).toString());
        });
    });
    return ret;
}


const requestListener = async function (req, res) {
    const path = req.url;
    const body = await readBody(req);

    let ret = '';
    try {
        if (path === '/display')
            ret = katex.renderToString(body, {...katexOpt, displayMode: true});
        else
            ret = katex.renderToString(body, katexOpt);

        res.writeHead(200);
        res.end(ret);
    }
    catch (e) {
        console.log(e);
        res.writeHead(500);
        res.end("");
    }
}

const server = http.createServer(requestListener);
server.listen(Number(process.argv[2]));

Python-markdown math extension

from xml.etree.ElementTree import Element
from markdown.inlinepatterns import InlineProcessor
from markdown.extensions import Extension

class MathPattern(InlineProcessor):
    def __init__(self, pattern, renderer):
        self.renderer = renderer
        super().__init__(pattern)


    def handleMatch(self, m, data):
        node = Element('span')
        node.text = self.renderer(m.group(2))
        return node, m.start(0), m.end(0)


class MathRenderExtension(Extension):
    def __init__(self, inline_renderer, display_renderer, *args, **kwargs):
        self.inline_renderer = inline_renderer
        self.display_renderer = display_renderer
        super().__init__(*args, **kwargs)

    def extendMarkdown(self, md):
        md.inlinePatterns.register(MathPattern(r'(?<!\\|\$)(\$)([^\$]+)(\$)', self.inline_renderer), 'math-inline', 185)
        md.inlinePatterns.register(MathPattern(r'(?<!\\)(\$\$)(([^\$])+)(\$\$)', self.display_renderer), 'math', 185)
        md.ESCAPED_CHARS.append('$')

Extension Setup

import subprocess
import time
import urllib.request as rq

katex = None
def katex_math_renderer(expr, display):
    global katex
    if not katex:
        katex = subprocess.Popen(['nodejs', 'katex.js', '8080'])
        time.sleep(2)
    res = rq.urlopen(f"http://localhost:8080/{'display' if display else ''}", data=expr.encode('utf-8'))
    return res.read().decode('utf-8')


md_config['extensions'].append(MathRenderExtension(
    lambda expr: katex_math_renderer(expr, False),
    lambda expr: katex_math_renderer(expr, True))
)
