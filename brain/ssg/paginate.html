def paginate(items, per_page=15, min_trailing=5):
    res = []
    for k in range(0,len(items),per_page):
        p = items[k:k+per_page]
        if len(p) > min_trailing:
            res.append(p)
        elif res:
            res[-1].extend(p)
        else:
            res.append(p)
    return res
